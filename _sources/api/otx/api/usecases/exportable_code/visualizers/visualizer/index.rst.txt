   .. rubric:: Example

   >>> predictions = inference_model.predict(frame)
   >>> annotation = prediction_converter.convert_to_annotation(predictions)
   >>> output = visualizer.draw(frame, annotation.shape, annotation.get_labels())
   >>> visualizer.show(output)
